# Optional: voice / microphone INPUT for Ember.
#
# pyaudio has no prebuilt wheel on macOS/Linux, so it compiles against the
# portaudio C library, which you must install first:
#   macOS:    brew install portaudio
#   Debian/Ubuntu: sudo apt install portaudio19-dev
#   Windows:  nothing extra (a prebuilt wheel is used)
#
# Then:  uv pip install -r requirements-voice.txt   (or pip install -r ...)
#
# Voice OUTPUT (text-to-speech) and everything else in Ember work WITHOUT this.
pyaudio>=0.2.14

# Optional: offline wake-word ("Hey Ember"). PocketSphinx runs the always-on wake
# listener locally (cheap + private). Without it Ember falls back to online Google
# Web Speech for wake detection, which works but makes a network call per listen.
pocketsphinx>=5.0.0

# Optional: natural neural text-to-speech with NO API key and no rate limits
# (Microsoft Edge voices). Select "Edge Neural TTS" in Settings ▸ Voice. Without it,
# Ember falls back to the system voice. Pure-Python; no system libs needed.
edge-tts>=6.1.0
